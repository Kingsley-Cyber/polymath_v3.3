from datetime import datetime

async def _upsert_document(
    driver: AsyncDriver,
    doc_id: str,
    corpus_id: str,
    user_id: str | None,
    file_id: str | None,
    filename: str,
    source_path: str | None = None,
    source_tier: str | None = None,
    chunk_count: int = 0,
    parent_count: int = 0,
    ghost_b_metrics: dict | None = None,
    schema_lens_id: str | None = None,
    file_metadata: dict | None = None,
) -> None:
    """
    Create or update a rich :Document node that serves as a strong cluster anchor.
    """
    async with driver.session() as session:
        await session.run(
            """
            MERGE (d:Document {doc_id: $doc_id})
            SET 
                d.corpus_id = $corpus_id,
                d.user_id = $user_id,
                d.file_id = $file_id,
                d.filename = $filename,
                d.source_path = $source_path,
                d.source_tier = $source_tier,
                d.chunk_count = $chunk_count,
                d.parent_count = $parent_count,
                d.ingested_at = $ingested_at,
                d.updated_at = datetime(),
                d.schema_lens_id = $schema_lens_id,
                d.ghost_b_metrics = $ghost_b_metrics,
                d.file_metadata = $file_metadata,
                
                -- Important for Brain View + frontend
                d.is_cluster_anchor = true,
                d.kind = "book",
                
                -- Optional: helps with future analytics
                d.node_type = "Document"
            """,
            doc_id=doc_id,
            corpus_id=corpus_id,
            user_id=user_id,
            file_id=file_id,
            filename=filename,
            source_path=source_path,
            source_tier=source_tier,
            chunk_count=chunk_count,
            parent_count=parent_count,
            ingested_at=datetime.utcnow(),
            schema_lens_id=schema_lens_id,
            ghost_b_metrics=ghost_b_metrics or {},
            file_metadata=file_metadata or {},
        )

cypher// Brain View: Book Anchors + Inter-Book Bridges
MATCH (d:Document)
WHERE d.corpus_id IN $corpus_ids 
  AND d.is_cluster_anchor = true

// Get basic stats for this document
OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)
WITH d, count(c) as actual_chunk_count

// Find bridges to other selected books via shared entities
OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)-[:MENTIONS]->(e:Entity)
OPTIONAL MATCH (e)-[r:RELATES_TO]->(e2:Entity)<-[:MENTIONS]-(c2:Chunk)<-[:HAS_CHUNK]-(d2:Document)
WHERE d2.corpus_id IN $corpus_ids 
  AND d2.is_cluster_anchor = true 
  AND d2.doc_id <> d.doc_id

WITH 
    d,
    actual_chunk_count,
    count(DISTINCT d2) as bridge_count,
    count(r) as total_bridge_relations,
    collect(DISTINCT {
        target_doc_id: d2.doc_id,
        target_filename: d2.filename,
        shared_entities: count(DISTINCT e),
        strength: count(r)
    }) as bridges_raw

// Aggregate bridge strength per target document
WITH d, actual_chunk_count, bridge_count, total_bridge_relations,
     apoc.coll.toSet(bridges_raw) as bridges

RETURN 
    d {
        .*,
        label: d.filename,
        node_kind: "Book",
        actual_chunk_count: actual_chunk_count,
        bridge_count: bridge_count,
        total_bridge_relations: total_bridge_relations
    } as document,
    bridges
ORDER BY bridge_count DESC, d.filename ASC
LIMIT 2000;   // safety cap

cypher// Brain View: Book Anchors + Bridge Strength (Multi-Corpus)
// Optimized for 1000-2000+ books

MATCH (d:Document)
WHERE d.corpus_id IN $corpus_ids 
  AND d.is_cluster_anchor = true

// Basic document stats
OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)
WITH d, count(c) as actual_chunk_count

// Bridge detection to other selected books
OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)-[:MENTIONS]->(e:Entity)
OPTIONAL MATCH (e)-[r:RELATES_TO]->(e2:Entity)<-[:MENTIONS]-(c2:Chunk)<-[:HAS_CHUNK]-(d2:Document)
WHERE d2.corpus_id IN $corpus_ids 
  AND d2.is_cluster_anchor = true 
  AND d2.doc_id <> d.doc_id

WITH 
    d,
    actual_chunk_count,
    count(DISTINCT d2) as bridge_count,
    count(r) as total_bridge_edges,
    collect(DISTINCT {
        target_doc_id: d2.doc_id,
        target_filename: COALESCE(d2.filename, d2.doc_id),
        strength: count(r),
        shared_entities: count(DISTINCT e)
    }) as raw_bridges

// Final aggregation
WITH d, actual_chunk_count, bridge_count, total_bridge_edges,
     apoc.coll.toSet(raw_bridges) as bridges

RETURN 
    d {
        .*,
        label: COALESCE(d.filename, d.doc_id),
        node_kind: "Book",
        actual_chunk_count: actual_chunk_count,
        bridge_count: bridge_count,
        total_bridge_edges: total_bridge_edges
    } as document,
    bridges
ORDER BY bridge_count DESC, d.filename ASC
LIMIT $limit;

async def get_brain_view(
    driver: AsyncDriver,
    corpus_ids: list[str],
    limit: int = 2000
) -> dict:
    query = """..."""  # paste the query above

    try:
        async with driver.session(database="neo4j") as session:  # or your DB name
            result = await session.run(
                query,
                corpus_ids=corpus_ids,
                limit=limit
            )
            
            records = await result.to_list()
            
            documents = []
            all_bridges = []
            
            for record in records:
                doc = record["document"]
                bridges = record["bridges"] or []
                
                documents.append(doc)
                for b in bridges:
                    all_bridges.append({
                        "source": doc["doc_id"],
                        "target": b["target_doc_id"],
                        "strength": b["strength"],
                        "shared_entities": b.get("shared_entities", 0)
                    })
            
            return {
                "documents": documents,
                "bridges": all_bridges,
                "meta": {
                    "corpus_count": len(set(d["corpus_id"] for d in documents)),
                    "total_documents": len(documents),
                    "limit_applied": limit
                }
            }
            
    except Exception as e:
        # Graceful degradation
        logger.error(f"Brain view query failed: {e}", exc_info=True)
        return {
            "documents": [],
            "bridges": [],
            "meta": {
                "error": str(e),
                "partial": True
            },
            "_error": True
        }

export const nodeReducer = (node: any) => {
  const kind = node.kind || node.node_kind || "Other";

  let visual = {
    color: NODE_COLORS[kind],
    size: NODE_SIZES[kind],
    mass: NODE_MASSES[kind],
    label: node.label || node.filename || node.name,
  };

  if (kind === "Book" || node.is_cluster_anchor) {
    visual = {
      ...visual,
      size: 18,                         // 🔁 Pt 3: 18 → 7 baseline + log2(bridge_count)
      mass: 35,                         // ✅ Pt 1
      borderColor: "#fcd34d",           // ⚠️ ignored by sigma default node program
      borderWidth: 4,                   // ⚠️ ignored
      haloColor: "#f59e0b44",           // ⚠️ ignored
      haloWidth: 18,                    // ⚠️ ignored
      labelSize: 16,                    // 🔁 Pt 3: → 13
      labelWeight: "700",               // 🔁 Pt 3: → "600"
    };
  }

  return { ...node, ...visual };
};

export const edgeReducer = (edge: any) => {
  const family = edge.relation_family || "WeakAssociation";
  const strength = edge.strength || 1;

  return {
    ...edge,
    color: EDGE_COLORS_BY_FAMILY[family],
    size: Math.min(1.2 + strength * 0.18, 5.5),   // ✅ Pt 5: NOW 0.35 + 0.05*strength,
                                                    //   cap 1.8px ("thin spaghetti").
    opacity: family === "WeakAssociation" ? 0.35 : 0.75 + (strength * 0.08),
                                                    // ✅ Pt 3 implemented via hexToRgba
                                                    //   in polymath-graph-adapter
    type: "arrow",                                  // ✅ honored
    curvature: 0.6,   // nice curved bridges        // ✅ EdgeCurveProgram v3
  };
};
