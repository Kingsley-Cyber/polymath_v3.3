export const EDGE_STYLES = {
  // Main function used by renderer
  getEdgeStyle: (family: string | null, predicate?: string) => {
    const safeFamily = (family as RelationFamily) || "WeakAssociation";
    const baseColor = EDGE_COLORS_BY_FAMILY[safeFamily];

    return {
      color: baseColor,
      size: getEdgeSize(safeFamily),
      type: "arrow" as const,           // or "line" / "dashed" for some families
      opacity: safeFamily === "WeakAssociation" || safeFamily === "Canonicalization" ? 0.45 : 0.85,
      // You can add label rendering logic here if you want predicate on hover
    };
  },
};

function getEdgeSize(family: RelationFamily): number {
  switch (family) {
    case "Conflict":
    case "Causal":
      return 2.8;
    case "Operational":
      return 2.0;
    case "Structural":
      return 1.8;
    case "WeakAssociation":
    case "Canonicalization":
      return 1.0;
    default:
      return 1.6;
  }
}

2. Brain View Query Design (Critical for Performance)
Never load the full graph for 2,000 books.
Top-level "Books-as-Clusters" query (fast anchor + bridge view):
cypherMATCH (d:Document)
WHERE d.corpus_id IN $corpus_ids 
  AND d.is_cluster_anchor = true

// Pre-compute bridge strength to other selected books
OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)-[:MENTIONS]->(e:Entity)
OPTIONAL MATCH (e)-[r:RELATES_TO]->(e2:Entity)<-[:MENTIONS]-(c2:Chunk)<-[:HAS_CHUNK]-(d2:Document)
WHERE d2.corpus_id IN $corpus_ids 
  AND d2 <> d
  AND d2.is_cluster_anchor = true

WITH d, 
     count(DISTINCT d2) as bridge_count,
     collect(DISTINCT {target: d2.doc_id, strength: count(r)}) as bridges

RETURN d {.*, label: d.filename, bridge_count} as document,
       bridges
ORDER BY bridge_count DESC
LIMIT 500;  // safety
Use Native Projection in GDS for any heavier community detection or similarity on the book level.
