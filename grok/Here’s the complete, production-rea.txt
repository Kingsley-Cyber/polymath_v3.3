Pythonfrom datetime import datetime
from neo4j import AsyncDriver
import logging
from typing import List, Dict, Any

logger = logging.getLogger(__name__)


async def get_book_drilldown(
    driver: AsyncDriver,
    doc_id: str,
    other_corpus_ids: List[str],
    limit: int = 350,
    max_hops: int = 2,
) -> Dict[str, Any]:
    """
    Load a single book's local neighborhood + cross-book bridges to other selected corpora.
    Used when user double-clicks a Book node in Brain View.
    """
    query = """
    // Drill-down: Single Book + Local Entities + Cross-Book Bridges
    MATCH (d:Document {doc_id: $doc_id})
    WHERE d.is_cluster_anchor = true

    // Local neighborhood
    OPTIONAL MATCH (d)-[:HAS_CHUNK]->(c:Chunk)
    OPTIONAL MATCH (c)-[:MENTIONS]->(e:Entity)
    OPTIONAL MATCH path = (e)-[r:RELATES_TO*1..$max_hops]->(e2:Entity)
    WHERE e2 <> e

    // Cross-book bridges
    OPTIONAL MATCH (e)-[rb:RELATES_TO]->(bridge:Entity)<-[:MENTIONS]-(:Chunk)<-[:HAS_CHUNK]-(d2:Document)
    WHERE d2.corpus_id IN $other_corpus_ids 
      AND d2.doc_id <> $doc_id
      AND d2.is_cluster_anchor = true

    WITH 
        d,
        collect(DISTINCT c) as chunks,
        collect(DISTINCT e) as local_entities,
        collect(DISTINCT r) as local_relations,
        collect(DISTINCT {
            bridge_entity: bridge,
            target_book_doc_id: d2.doc_id,
            target_book_filename: COALESCE(d2.filename, d2.doc_id),
            strength: count(rb)
        }) as raw_bridges

    WITH d, chunks, local_entities, local_relations,
         apoc.coll.toSet(raw_bridges) as cross_book_bridges

    RETURN 
        d {
            .*,
            label: COALESCE(d.filename, d.doc_id),
            node_kind: "Book"
        } as anchor,
        local_entities,
        local_relations,
        cross_book_bridges
    LIMIT $limit;
    """

    try:
        async with driver.session() as session:
            result = await session.run(
                query,
                doc_id=doc_id,
                other_corpus_ids=other_corpus_ids,
                limit=limit,
                max_hops=max_hops
            )
            
            record = await result.single()

            if not record:
                return {
                    "anchor": None,
                    "local_entities": [],
                    "local_relations": [],
                    "cross_book_bridges": [],
                    "meta": {"found": False}
                }

            anchor = record["anchor"]
            local_entities = record["local_entities"] or []
            local_relations = record["local_relations"] or []
            cross_book_bridges = record["cross_book_bridges"] or []

            return {
                "anchor": anchor,
                "local_entities": local_entities,
                "local_relations": local_relations,
                "cross_book_bridges": cross_book_bridges,
                "meta": {
                    "found": True,
                    "local_entity_count": len(local_entities),
                    "bridge_count": len(cross_book_bridges),
                    "limit": limit
                }
            }

    except Exception as e:
        logger.error(f"Drilldown query failed for doc_id={doc_id}: {e}", exc_info=True)
        return {
            "anchor": None,
            "local_entities": [],
            "local_relations": [],
            "cross_book_bridges": [],
            "meta": {
                "found": False,
                "error": str(e),
                "partial": True
            },
            "_error": True
        }

async def _upsert_entity_and_mention(
    driver: AsyncDriver,
    entity_data: dict,
    chunk_id: str,
    corpus_id: str,
    doc_id: str
):
    query = """
    // Upsert Entity with strong uniqueness
    MERGE (e:Entity {entity_id: $entity_id})
    ON CREATE SET 
        e.normalized_name = $normalized_name,
        e.created_at = datetime()
    SET 
        e.display_name = $display_name,
        e.entity_type = $entity_type,
        e.canonical_name = $normalized_name,   // backward compat
        e.updated_at = datetime()

    // Link to Chunk
    WITH e
    MATCH (c:Chunk {chunk_id: $chunk_id})
    MERGE (c)-[m:MENTIONS]->(e)
    SET 
        m.confidence = $confidence,
        m.evidence_phrase = $evidence_phrase,
        m.corpus_id = $corpus_id,
        m.doc_id = $doc_id
    """

    params = {
        "entity_id": entity_data.get("entity_id") or f"ent_{entity_data['normalized_name']}",
        "normalized_name": entity_data["normalized_name"],
        "display_name": entity_data.get("surface_form") or entity_data["normalized_name"],
        "entity_type": entity_data["entity_type"],
        "confidence": entity_data.get("confidence", 0.8),
        "evidence_phrase": entity_data.get("evidence_phrase"),
        "chunk_id": chunk_id,
        "corpus_id": corpus_id,
        "doc_id": doc_id,
    }

    async with driver.session() as session:
        await session.run(query, params)
